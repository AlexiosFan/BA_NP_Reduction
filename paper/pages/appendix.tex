\chapter{Karp's Definition of NP-hard problems mentioned}
In this part, we present Karp's original definition of the NP-hard problems included in this work.
Readers are encouraged to compare them with the definitions in the previous chapters.
\problem{Satisfiability}{Clauses $C_1, C_2, ..., C_p$}{Is there a set 
  $S \subset \{x_1, x_2, ..., x_n, \overline{x_1}, \overline{x_2}, ..., \overline{x_n}\}$ s.t. 
\begin{enumerate}
    \item S does not contain a complementary pair of literals.
    \item $S \cap C_k \neq \emptyset$, for all $k$ from $1$ to $p$.
\end{enumerate}}
\problem{Exact Cover}{
    Family $\{S_j\}$ of subsets of a set $\{u_i | i = 1, 2, ... ,t\}$
}{
    Is there a subfamily $\{T_h\} \subseteq \{S_j\}$ such that the sets $T_h$ are disjoint 
    and $\bigcup T_h = \bigcup S_j = \{u_i | i = 1, 2, ... ,t\}$?
}
\problem{(Exact) Hitting Set}{Family $\{U_j\}$ of subsets of $\{s_j | j = 1, 2, ..., r\}$}{
    Is there a set $W$ s.t. for each $i$, $|W \cap U_i| = 1$?
}
\problem{Subset Sum, Knapsack in Karp's definition}{
    $(a_1, a_2, ..., a_n, b) \in Z^{n+1}$
}{Does $\sum a_j x_j = b$ have a zero-one solution?}
\problem{Partition}{
    $(c_1, ..., c_s) \in Z^s$
}{
    Is there a subset $I \subseteq \{1, 2, ..., s\}$ s.t. $\sum_{h \in I}c_h = \sum_{h \not\in I} c_h$?
}
\problem{Zero-One Integer Programming}{
    Integer Matrix C and integer vector d
}{Does there exist a zero-one vector x s.t. $Cx = d$?}
